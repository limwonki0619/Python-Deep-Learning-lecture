#   

model = Sequential([
    
    Conv2D(32, (3,3), padding='same', input_shape=(32, 32, 3), activation='elu', kernel_regularizer=l2(0.001)),
    BatchNormalization(),
    Conv2D(32, (3,3), padding='same', activation='elu'), # elu, Leaky Relu(LRelu), Randomized Relu(RRelu)
    BatchNormalization(),
    MaxPooling2D(pool_size=(2,2)),
    Dropout(.2),
    
    Conv2D(64, (3,3), padding='same', activation='elu', kernel_regularizer=l2(0.001)), # L2  
    BatchNormalization(),
    Conv2D(64, (3,3), padding='same', activation='elu', kernel_regularizer=l2(0.001)),
    BatchNormalization(),
    MaxPooling2D(pool_size=(2,2)),
    Dropout(.3),
    
    Conv2D(128, (3,3), padding='same', activation='elu', kernel_regularizer=l2(0.001)),
    BatchNormalization(),
    Conv2D(128, (3,3), padding='same', activation='elu', kernel_regularizer=l2(0.001)),
    BatchNormalization(),
    MaxPooling2D(pool_size=(2,2)),
    Dropout(.4),     
    
    Flatten(),
    Dense(128, activation='elu', kernel_regularizer=l2(0.001)),
    Dropout(.5), 
    Dense(10, activation='softmax')
])
